# jm-shuffler
card shuffler

```javascript
// nodejs
const Shuffler = require('jm-shuffler').Shuffler

var shuffler = new Shuffler()
shuffler.shuffle([1, 2, 3])

```

```javascript
// es6
import shuff from 'jm-shuffler'
let Shuffler = shuff.Shuffler

let shuffler = new Shuffler()
shuffler.shuffle([1, 2, 3])

```
